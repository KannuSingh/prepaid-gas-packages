## Multi-Chain Strategy

### Network Detection

```typescript
// utils/network-detection.ts
import { useChainId } from 'wagmi'

export function useNetworkConfig() {
  const chainId = useChainId()
  
  const isSupported = [84532, 8453].includes(chainId)
  const isTestnet = chainId === 84532
  
  return {
    chainId,
    isSupported,
    isTestnet,
    networkName: getNetworkName(chainId)
  }
}

function getNetworkName(chainId: number): string {
  switch (chainId) {
    case 84532: return 'Base Sepolia'
    case 8453: return 'Base'
    default: return 'Unsupported'
  }
}
```

### Cross-Chain Privacy

Privacy considerations across multiple networks:

```typescript
// Different identities per network for maximum privacy
const identityManager = new Map<number, Identity>()

function getNetworkIdentity(chainId: number): Identity {
  if (!identityManager.has(chainId)) {
    identityManager.set(chainId, new Identity())
  }
  return identityManager.get(chainId)!
}
```
